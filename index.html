function processarExcelXPInteligente(file) { const reader = new FileReader(); reader.onload = function(e) { const data = new Uint8Array(e.target.result); const workbook = XLSX.read(data, { type: 'array' }); const sheet = workbook.Sheets[workbook.SheetNames[0]]; const json = XLSX.utils.sheet_to_json(sheet, { header: 1, defval: "" }); // Estrutura para acumular valores let portfolio = { acoes: 0, rendaFixa: { prefixada: 0, posFixada: 0, inflacao: 0, outros: 0 }, fundos: { pos: 0, inflacao: 0, multimercado: 0, acoes: 0, cambial: 0 }, previdencia: { ficfia: 0, ficfim: 0, outros: 0 }, outros: 0 }; let totalGeral = 0; let secoes = {}; // Armazena o range de linhas de cada seção // ========== PASSO 1: IDENTIFICAR SEÇÕES POR PADRÕES ========== let secaoAtual = null; let dentroTabela = false; for (let i = 0; i < json.length; i++) { const linha = json[i]; if (!linha || linha.length === 0) continue; const primeiraCelula = (linha[0] || "").toString().trim(); const textoLinha = linha.join(" ").toLowerCase(); // Detecta títulos de seção (padrão: "XX,XX%|Nome da Seção") const matchSecao = primeiraCelula.match(/(\d+[\.,]?\d*)%\|(.+)/); if (matchSecao) { const percentual = parseFloat(matchSecao[1].replace(',', '.')); const nomeSecao = matchSecao[2].toLowerCase(); // Classifica a seção por heurística let tipo = null; if (nomeSecao.includes("ação") || nomeSecao.includes("acoes") || nomeSecao.includes("stock")) tipo = "acoes"; else if (nomeSecao.includes("prefixada") || nomeSecao.includes("pre")) tipo = "prefixada"; else if (nomeSecao.includes("pós") || nomeSecao.includes("pos") || nomeSecao.includes("cdi")) tipo = "posFixada"; else if (nomeSecao.includes("inflação") || nomeSecao.includes("ipca")) tipo = "inflacao"; else if (nomeSecao.includes("fundo")) tipo = "fundos"; else if (nomeSecao.includes("previdência") || nomeSecao.includes("prev")) tipo = "previdencia"; if (tipo) { secoes[i] = { tipo, nome: matchSecao[2], percentualDoTotal: percentual, linhaInicio: i }; secaoAtual = i; dentroTabela = true; } continue; } // Detecta subseções dentro de fundos (ex: "16,50%|Fundos de Renda Fixa Pós-Fixado") const matchSubSecao = primeiraCelula.match(/(\d+[\.,]?\d*)%\|(.+)/); if (matchSubSecao && secaoAtual && secoes[secaoAtual]?.tipo === "fundos") { const nomeSub = matchSubSecao[2].toLowerCase(); let subTipo = null; if (nomeSub.includes("pós") || nomeSub.includes("pos") || nomeSub.includes("di")) subTipo = "pos"; else if (nomeSub.includes("inflação") || nomeSub.includes("ipca")) subTipo = "inflacao"; else if (nomeSub.includes("multimercado") || nomeSub.includes("multi")) subTipo = "multimercado"; else if (nomeSub.includes("ação") || nomeSub.includes("acoes") || nomeSub.includes("renda variável")) subTipo = "acoes"; else if (nomeSub.includes("cambial") || nomeSub.includes("dólar")) subTipo = "cambial"; if (subTipo) { secoes[i] = { tipo: "subfundos", subTipo, nome: matchSubSecao[2], percentualDoTotal: parseFloat(matchSubSecao[1].replace(',', '.')), linhaInicio: i }; } } } // ========== PASSO 2: EXTRAIR VALORES DAS SEÇÕES ========== for (let i = 0; i < json.length; i++) { const linha = json[i]; if (!linha || linha.length === 0) continue; // Procura por valores monetários no final da linha (padrão XP: "... R$ 123.456,78") const ultimaColuna = linha[linha.length - 1]; const penultimaColuna = linha[linha.length - 2]; let valor = 0; // Tenta extrair valor da última coluna if (ultimaColuna && typeof ultimaColuna === 'string' && ultimaColuna.includes('R$')) { valor = extrairValorMonetario(ultimaColuna); } else if (penultimaColuna && typeof penultimaColuna === 'string' && penultimaColuna.includes('R$')) { valor = extrairValorMonetario(penultimaColuna); } else { valor = extrairValorMonetario(ultimaColuna); } if (valor === 0) continue; // Verifica se esta linha é um título de seção que já identificamos for (let [idx, secao] of Object.entries(secoes)) { if (parseInt(idx) === i && !secao.valorExtraido) { secao.valorExtraido = true; if (secao.tipo === "acoes") { portfolio.acoes = valor; } else if (secao.tipo === "prefixada") { portfolio.rendaFixa.prefixada = valor; } else if (secao.tipo === "posFixada") { portfolio.rendaFixa.posFixada = valor; } else if (secao.tipo === "inflacao") { portfolio.rendaFixa.inflacao = valor; } else if (secao.tipo === "fundos") { portfolio.totalFundos = valor; } else if (secao.tipo === "previdencia") { portfolio.totalPrevidencia = valor; } break; } } // ========== PASSO 3: DENTRO DA TABELA DE FUNDOS, EXTRAIR COTAS ========== if (secoes[i]?.tipo === "subfundos") { // A linha seguinte ao título contém os cabeçalhos, depois vêm os dados let linhaDados = i + 3; // Pula cabeçalho let somatorio = 0; let contador = 0; while (json[linhaDados] && json[linhaDados][0] && !json[linhaDados][0].toString().includes('%')) { const linhaFundo = json[linhaDados]; const valorFundo = extrairValorMonetario(linhaFundo[linhaFundo.length - 1]); if (valorFundo > 0 && valorFundo < 10000000) { const subTipo = secoes[i].subTipo; if (subTipo === "pos") portfolio.fundos.pos += valorFundo; else if (subTipo === "inflacao") portfolio.fundos.inflacao += valorFundo; else if (subTipo === "multimercado") portfolio.fundos.multimercado += valorFundo; else if (subTipo === "acoes") portfolio.fundos.acoes += valorFundo; else if (subTipo === "cambial") portfolio.fundos.cambial += valorFundo; somatorio += valorFundo; contador++; } linhaDados++; } // Se conseguiu somar valores individuais, usa eles. Senão, usa o percentual do total de fundos if (somatorio > 0 && portfolio.totalFundos > 0 && Math.abs(somatorio - (portfolio.totalFundos * secoes[i].percentualDoTotal / 100)) > 1000) { // Valores individuais são mais confiáveis } } // ========== PASSO 4: EXTRAIR FUNDOS DE PREVIDÊNCIA ========== if (secoes[i]?.tipo === "previdencia") { let linhaPrev = i + 3; // Pula cabeçalho while (json[linhaPrev] && json[linhaPrev][0] && !json[linhaPrev][0].toString().includes('Custódia')) { const linhaFundo = json[linhaPrev]; const nomeFundo = (linhaFundo[0] || "").toString().toLowerCase(); const valorFundo = extrairValorMonetario(linhaFundo[linhaFundo.length - 1]); if (valorFundo > 0 && valorFundo < 10000000) { if (nomeFundo.includes("ficfia")) portfolio.previdencia.ficfia += valorFundo; else if (nomeFundo.includes("ficfim")) portfolio.previdencia.ficfim += valorFundo; else portfolio.previdencia.outros += valorFundo; } linhaPrev++; } } } // ========== PASSO 5: AGREGAR TOTAIS FINAIS ========== let totalPos = portfolio.rendaFixa.posFixada + portfolio.fundos.pos + portfolio.previdencia.outros; let totalPre = portfolio.rendaFixa.prefixada; let totalInflacao = portfolio.rendaFixa.inflacao + portfolio.fundos.inflacao; let totalMulti = portfolio.fundos.multimercado + portfolio.previdencia.ficfim; let totalRV = portfolio.acoes + portfolio.fundos.acoes + portfolio.previdencia.ficfia; let patrimonio = totalPos + totalPre + totalInflacao + totalMulti + totalRV; // Fallback: tenta pegar o "Total investido" do cabeçalho for (let i = 0; i < json.length; i++) { const linha = json[i]; if (linha && linha[0] && linha[0].toString().includes("Total investido")) { patrimonio = extrairValorMonetario(linha[1]); if (patrimonio > 0) break; } } // Calcula percentuais const pcts = { pos: (totalPos / patrimonio) * 100, pre: (totalPre / patrimonio) * 100, inflacao: (totalInflacao / patrimonio) * 100, multi: (totalMulti / patrimonio) * 100, rv: (totalRV / patrimonio) * 100 }; // Preenche os inputs if (inputsAtuais["Renda Fixa Pós"]) inputsAtuais["Renda Fixa Pós"].value = pcts.pos.toFixed(2); if (inputsAtuais["Renda Fixa Pré"]) inputsAtuais["Renda Fixa Pré"].value = pcts.pre.toFixed(2); if (inputsAtuais["Renda Fixa Inflação"]) inputsAtuais["Renda Fixa Inflação"].value = pcts.inflacao.toFixed(2); if (inputsAtuais["Multiestratégia"]) inputsAtuais["Multiestratégia"].value = pcts.multi.toFixed(2); if (inputsAtuais["Renda Variável"]) inputsAtuais["Renda Variável"].value = pcts.rv.toFixed(2); atualizarTotal(); // Exibe relatório detalhado const detalhesDiv = document.getElementById("detalhesImportacao"); detalhesDiv.innerHTML = ` ✅ Importação inteligente concluída!

💰 Patrimônio total: R$ ${patrimonio.toLocaleString('pt-BR')}

📊 Valores por classe (R$):
• Renda Fixa Pós: R$ ${totalPos.toLocaleString('pt-BR')} (${pcts.pos.toFixed(2)}%)
• Renda Fixa Pré: R$ ${totalPre.toLocaleString('pt-BR')} (${pcts.pre.toFixed(2)}%)
• Renda Fixa Inflação: R$ ${totalInflacao.toLocaleString('pt-BR')} (${pcts.inflacao.toFixed(2)}%)
• Multiestratégia: R$ ${totalMulti.toLocaleString('pt-BR')} (${pcts.multi.toFixed(2)}%)
• Renda Variável: R$ ${totalRV.toLocaleString('pt-BR')} (${pcts.rv.toFixed(2)}%)
🔍 Composição detalhada:
• Ações diretas: R$ ${portfolio.acoes.toLocaleString('pt-BR')}
• Renda Fixa Prefixada: R$ ${portfolio.rendaFixa.prefixada.toLocaleString('pt-BR')}
• Renda Fixa Pós: R$ ${portfolio.rendaFixa.posFixada.toLocaleString('pt-BR')}
• Renda Fixa Inflação: R$ ${portfolio.rendaFixa.inflacao.toLocaleString('pt-BR')}
• Fundos Pós: R$ ${portfolio.fundos.pos.toLocaleString('pt-BR')}
• Fundos Inflação: R$ ${portfolio.fundos.inflacao.toLocaleString('pt-BR')}
• Fundos Multimercado: R$ ${portfolio.fundos.multimercado.toLocaleString('pt-BR')}
• Fundos Ações: R$ ${portfolio.fundos.acoes.toLocaleString('pt-BR')}
• Previdência FICFIA (RV): R$ ${portfolio.previdencia.ficfia.toLocaleString('pt-BR')}
• Previdência FICFIM (Multi): R$ ${portfolio.previdencia.ficfim.toLocaleString('pt-BR')}
• Previdência Outros: R$ ${portfolio.previdencia.outros.toLocaleString('pt-BR')}
 `; detalhesDiv.classList.add("show"); alert(`✅ Importação concluída!\n\nPatrimônio: R$ ${patrimonio.toLocaleString('pt-BR')}\n\nPós: ${pcts.pos.toFixed(1)}%\nPré: ${pcts.pre.toFixed(1)}%\nInflação: ${pcts.inflacao.toFixed(1)}%\nMulti: ${pcts.multi.toFixed(1)}%\nRV: ${pcts.rv.toFixed(1)}%`); }; reader.readAsArrayBuffer(file); }
