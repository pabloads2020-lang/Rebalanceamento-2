async function extrairDadosPDF(file) { const ab = await file.arrayBuffer(); const pdf = await pdfjsLib.getDocument({ data: ab }).promise; let raw = []; for (let p = 1; p <= pdf.numPages; p++) { const tc = await (await pdf.getPage(p)).getTextContent(); tc.items.forEach(it => raw.push(it.str)); } function norm(s) { return s .replace(/\uFB00/g, 'ff') .replace(/\uFB01/g, 'fi') .replace(/\uFB02/g, 'fl') .replace(/\uFB03/g, 'ffi') .replace(/\uFB04/g, 'ffl') .replace(/\u00A0/g, ' ') .trim(); } // 🔥 NORMALIZAÇÃO FORTE (resolve inflação e qualquer acento) function normalizarTexto(s) { return s .normalize("NFD") .replace(/[\u0300-\u036f]/g, "") .toLowerCase() .trim(); } const T = raw.map(norm).filter(s => s.length > 0); dbg('Tokens: ' + T.length); function lerPct(i) { if (T[i] !== '(') return null; const a = T[i + 1] || '', sep = T[i + 2] || '', b = T[i + 3] || '', fim = T[i + 4] || ''; if (sep === ',' && fim.endsWith('%)') && /^\d+$/.test(a) && /^\d+$/.test(b)) return parseFloat(a + '.' + b); return null; } function lerValor(i) { if (T[i] !== 'R') return null; let j = i + 2; let numero = ''; while (j < T.length && j < i + 20) { const tk = T[j]; if (/^\d+$/.test(tk)) { numero += tk; j++; } else if (tk === '.') { numero += '.'; j++; } else if (tk === ',') { const prox = T[j + 1] || ''; if (/^\d{2}$/.test(prox)) { numero += ',' + prox; j += 2; break; } else break; } else break; } if (!numero || !numero.includes(',')) return null; return parseFloat(numero.replace(/\./g, '').replace(',', '.')); } // 🔥 NOVO MATCH MUITO MAIS ROBUSTO const NOMES = [ { match: t => t.includes("pos fixado"), destino: "Pós Fixado" }, { match: t => t.includes("pre fixado"), destino: "Pré Fixado" }, { match: t => t.includes("infl"), destino: "Inflação" }, // ← corrigido de verdade { match: t => t.includes("multimercado"), destino: "Multimercado" }, { match: t => t.includes("renda variavel"), destino: "Renda Variável" }, { match: t => t.includes("alternativo"), destino: "Renda Variável" } ]; const res = {}; for (let i = 0; i < T.length; i++) { const token = normalizarTexto(T[i]); for (const { match, destino } of NOMES) { if (!match(token)) continue; let pi = -1; for (let k = i + 1; k <= i + 4 && k < T.length; k++) { if (T[k] === '(') { pi = k; break; } } if (pi === -1) continue; const pct = lerPct(pi); if (pct === null) continue; let ri = -1; for (let k = pi + 5; k <= pi + 10 && k < T.length; k++) { if (T[k] === 'R') { ri = k; break; } } const val = ri !== -1 ? lerValor(ri) : null; if (!res[destino]) { res[destino] = { pct: 0, val: 0 }; } res[destino].pct += pct; if (val) res[destino].val += val; dbg(`[OK] ${token} → ${destino}: ${pct}% | R$${val}`); break; } } // Recalcular percentuais se necessário const totVal = CLASSES.reduce((s, c) => s + (res[c]?.val || 0), 0); if (totVal > 0) { for (const c of CLASSES) { if (res[c] && res[c].val > 0 && !res[c].pct) { res[c].pct = (res[c].val / totVal) * 100; dbg(`[Recalc] ${c}: ${res[c].pct.toFixed(2)}%`); } } } // Patrimônio let patrimonio = totVal; for (let i = 0; i < T.length - 3; i++) { if (/PATRIM/i.test(T[i]) && /TOTAL/i.test(T[i + 1]) && /BRUTO/i.test(T[i + 2])) { for (let k = i + 3; k < Math.min(i + 10, T.length); k++) { if (T[k] === 'R') { const v = lerValor(k); if (v) { patrimonio = v; dbg(`[Patr] R$${v}`); break; } } } break; } } return { res, patrimonio }; }
